USER SCAN SUMMARY

CREATE OR REPLACE VIEW user_scan_summary_final AS
SELECT
  org,
  actor_email,
  sum(scans)        AS total_scans,
  sum(detections)   AS total_detections,
  sum(blocks)       AS total_blocks,
  max(last_activity) AS last_activity
FROM user_scan_summary
GROUP BY org, actor_email;

USER CATEGORY COUNTS

CREATE VIEW user_category_counts_final AS
SELECT
  org,
  actor_email,
  category,
  sum(cnt) AS count
FROM user_category_counts
GROUP BY org, actor_email, category;


SOURCE BREAKDOWN

CREATE VIEW source_stats_final AS
SELECT
  org,
  source,
  sum(cnt) AS count
FROM source_stats
GROUP BY org, source;


POLICY EFFECTIVENESS

CREATE VIEW policy_effectiveness_final AS
SELECT
  org,
  policy,
  sum(triggered) AS triggered,
  sum(blocked)   AS blocked,
  round(sum(blocked) / sum(triggered), 3) AS block_ratio
FROM policy_effectiveness
GROUP BY org, policy;

SCAN TIMELINE

SELECT
